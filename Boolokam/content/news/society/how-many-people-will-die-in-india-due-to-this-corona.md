<strong>ഈ കൊറോണോ മൂലം ഇന്ത്യയിൽ എത്ര പേർ മരിക്കും</strong>

ആർക്കും അത് മുൻകൂട്ടി പറയുവാൻ സാധിക്കില്ല.പക്ഷെ ഒരു കാര്യം ഉറപ്പാണ് ഇന്ത്യയിൽ ലോക്ക് ഡൌൺ മൂലം ഉണ്ടാകുന്ന പട്ടിണിയും കൂട്ടത്തോടെയുള്ള പാലായനവും ,കലാപങ്ങളും ,അപകടങ്ങളും മൂലം ഉണ്ടാവുന്ന അത്ര മരണം കൊറോണോ മൂലം ഉണ്ടാവാൻ സാധ്യതയില്ല.ഇതുമായി അൽപ്പം മാറിയൊരു വിഷയമാണ് ഇപ്പോൾ പറയുന്നത്.ഇപ്പോൾ റോഡിലേയ്ക്ക് ഇറങ്ങുന്നവർക്ക് പ്രതെയ്കിച്ചു വണ്ടിയില്ലാതെ ഇറങ്ങുന്നവർക്ക് ഒരു കാര്യം മനസ്സിലാവും .ഇത് വരെ വളരെ ബുദ്ധിമുട്ടിയും പോലീസിനെ ഭയന്നും അനുസരിച്ചിരുന്ന ട്രാഫിക്ക് നിയുമങ്ങൾ എല്ലാം ഇപ്പോൾ കാറ്റിൽ പരത്തിയിരിക്കുന്നു.അരാജകത്വം വന്ന പ്രതീതിയാണ് റോഡിൽ.അമിത വേഗമാണ് എവിടെയും.

റോഡ് സുരക്ഷയെക്കുറിച്ചുള്ള ലോകാരോഗ്യ സംഘടനയുടെ ആഗോള റിപ്പോർട്ട് അനുസരിച്ച്, ലോകത്തെ റോഡ് അപകടവുമായി ബന്ധപ്പെട്ട മരണങ്ങളിൽ ഏകദേശം 11% ഇന്ത്യയിലാണ് . 2018 ൽ മാത്രം 4,67,044 റോഡപകടങ്ങൾ ഇവിടെ റിപ്പോർട്ട് ചെയ്തിട്ടുണ്ട്, 1,51,417 പേർ കൊല്ലപ്പെടുകയും 4,69,418 പേർക്ക് പരിക്കേൽക്കുകയും ചെയ്തു. ഓരോ മിനിറ്റിലും നാല് മരണം.
അംഗം ഭംഗം,കോമ സ്റെജ്ജുകൾ ,ജീവിത കാലം മുഴുവൻ കിടപ്പിലാവുന്നവർ,പട്ടിണി,ദുരിതം ,തീരാ വേദന ,അനാഥ ജീവിതങ്ങൾ എല്ലാം വേറെ...

നമ്മുടെ റോഡ് അപകടങ്ങൾക്ക് പ്രധാന കാരണങ്ങൾ ഇവയാണ് -അഹന്ത,അക്ജ്ഞത ,അസഹിഷ്‌ണുത.
കന്നി മൂലയിൽ ടോയ്‌ലറ്റ്‌ വയ്ക്കാതിരിക്കാനും ,കിഴക്കോട്ട് തല വെച്ച് കിടന്ന് ഉറങ്ങാനും,നെഗറ്റിവ് എനെർജ്ജി ഒഴിവാക്കാനും,ദൈവങ്ങളെ രക്ഷിക്കുവാനും,രാഹുവിനെ പൂട്ടനുമെല്ലാം ഉള്ള പെടാ പാടിന്റെ 1 % ശ്രദ്ധ റോഡ് അപകടങ്ങൾ ഒഴിവാക്കുവാൻ ആളുകൾ കാണിക്കുന്നില്ല.മാധ്യമങ്ങളും,സന്നദ്ധ സംഘടനകളും,സർക്കാർ സംവിധാനങ്ങളും ട്രാഫിക്ക് നിയമങ്ങൾ കർശനമായി നടപ്പാക്കുന്ന കാര്യത്തിൽ ഊന്നൽ നൽകുന്നില്ല എന്നു പറയാതെ വയ്യ.

അപടകങ്ങൾ റിപ്പോർട്ട് ചെയ്യുമ്പോൾ "ട്രാഫിക്ക് നിയമ ലംഘനം ഇന്നലെ കോട്ടയത്തു 10 മരണം " എന്ന രീതിയിൽ റിപ്പോർട്ട് ചെയ്തിരുന്നെങ്കിൽ തന്നെ ആളുകളിൽ ഒരു മാനസിക പരിവർത്തനത്തിന് സാധ്യത തെളിഞ്ഞേനെ .വീട്ടിൽ നിന്ന് രാവിൽ ദൂരെ യാത്രക്ക് പോകുന്ന മക്കൾളോട് അമ്മമാർ പറയുന്ന ഒരു കാര്യമുണ്ട് -പ്രാർത്ഥിച്ചിട്ട് ആണോ ഇറങ്ങിയത്? ഇതിന് പകരം "സീറ്റ് ബെൽറ്റ് ഇട്ട് ,ഹെൽമെറ്റ് വച്ചു,മറ്റു ട്രാഫിക്ക് നിയമങ്ങൾ അനുസരിച്ചു വണ്ടി ഓടിക്കണേ മക്കളെ എന്നു പറയാൻ ശ്രമിച്ചാൽ നല്ലതായിരുന്നു.
പുതിയ വണ്ടി വാങ്ങിയാൽ അത് പൂജിക്കുവാനും,വെഞ്ചരിക്കുവാനും,റിയർ വ്യൂ മിററിൽ വിശുദ്ധ വസ്തുക്കൾ തൂക്കാനും ശ്രമിക്കുന്നതിന്റ ഒരു പങ്ക് സമയം മാന്യമായി ,സുരക്ഷിതമായി എങ്ങനെ വണ്ടി ഓടിക്കണം എന്ന് ഒരു ലഖു ലേഖ എങ്കിലും വാങ്ങി വായിക്കുവാൻ ചിലവഴിച്ചിരുന്നെങ്കിൽ.

വീട്ടിൽ നിന്ന് ഇറങ്ങിയാൽ എത്ര പേര് തിരിച്ചു വരും എന്ന് അറിയുവാൻ സാധിക്കാത്ത ഒരു അഡ്വെഞ്ചർ സ്പോർട്സ് ആണ് ഇന്ത്യൻ നിരത്തിലെ ഡ്രൈവിങ്.ഇത് മനസിലാക്കി ഓരോ ദിവസവും വണ്ടി സ്റ്റാർട്ട് ചെയ്യുക.റോഡുകൾ തോറും സർവയിലന്സ് ക്യാമറകൾ വയ്ക്കുകയും നിയമങ്ങൾ തെറ്റിക്കുന്നവർക്ക് കൃത്യമായി പിഴ വീട്ടിലേയ്ക്ക് അയക്കുകയും ചെയ്യുക-ഈ പിഴയിൽ നിന്ന് തന്നെ ക്യാമറ സ്ഥാപിക്കുവാനുള്ള പണം കണ്ടെത്തുകയും ചെയ്യാവുന്നതാണ്.സ്ക്കൂൾ തലം മുതൽ കുട്ടികളെ റോഡ് നിയമങ്ങളെ കുറിച്ച് വിശദമായി തന്നെ പഠിപ്പിക്കുക.

റോഡിൽ മാന്യമായി വണ്ടി ഓടിക്കാത്തത് തികച്ചും അപരിഷ്‌കൃതമായ നടപടിയാണ് എന്ന ബോധം കുട്ടികളിൽ ഉണ്ടാക്കിയെടുക്കുക. ക്രമേണ നിയമ ലംഘനം നടത്തുന്നവരോട് കുട്ടികൾക്ക് ഒരു അവക്ഞ്ഞ ഉണ്ടാവുകയും അവർ തന്നെ മാതാപിതാക്കന്മാരെ തിരുത്തുകയും ചെയ്യും.റോഡിൽ നിയമങ്ങൾ പാലിക്കേണ്ടതിന്റെ ആവശ്യകതെയെ കുറിച്ചുള്ള ക്‌ളാസ്സുകൾ മത പഠനങ്ങളുടെ കൂടെ ചേർക്കുക. ജീവൻ പോയാൽ പിന്നെ തെക്കോട്ട് തല വച്ച് ഉറങ്ങുക എന്ന ഒരു പോം വഴിയേ ഉള്ളു.
